/* ============================================================ MOSTRAR SELECTOR DE CITA SOLO CUANDO EL FORMULARIO ESTÁ LLENO ============================================================ */ function mostrarSelectorCita() { const nombre = document.getElementById("nombre").value.trim(); const telefono = document.getElementById("telefono").value.trim(); const referencia = document.getElementById("referencia").value.trim(); const direccion = document.getElementById("direccion").value.trim(); if (!nombre || !telefono || !referencia || !direccion) { alert("Por favor, completa todos los datos antes de fijar la cita."); return; } document.getElementById("bloqueCita").style.display = "block"; window.scrollTo(0, document.body.scrollHeight); } /* ============================================================ 1) CUANDO EL CLIENTE ELIGE UNA HORA → ABRIR GOOGLE CALENDAR ============================================================ */ function onCitaSeleccionada() { const select = document.getElementById("citaElegida"); const confirmarBtn = document.getElementById("confirmarBtn"); if (!select.value) { confirmarBtn.style.display = "none"; return; } // Fecha de hoy en formato YYYYMMDD const hoy = new Date(); const yyyy = hoy.getFullYear(); const mm = String(hoy.getMonth() + 1).padStart(2, "0"); const dd = String(hoy.getDate()).padStart(2, "0"); const fecha = `${yyyy}${mm}${dd}`; // Horarios disponibles const horas = { "Hoy a las 10:00": ["100000", "110000"], "Hoy a las 11:30": ["113000", "123000"], "Hoy a las 13:00": ["130000", "140000"], "Hoy a las 16:30": ["163000", "173000"], "Hoy a las 18:00": ["180000", "190000"] }; const [inicio, fin] = horas[select.value]; const referencia = document.getElementById("referencia").value; const direccion = document.getElementById("direccion").value; const titulo = encodeURIComponent(`Visita inmueble ${referencia}`); const detalles = encodeURIComponent("Visita con cliente"); const loc = encodeURIComponent(direccion); const url = `https://calendar.google.com/calendar/render?action=TEMPLATE` + `&text=${titulo}` + `&dates=${fecha}T${inicio}/${fecha}T${fin}` + `&details=${detalles}` + `&location=${loc}`; window.open(url, "_blank"); confirmarBtn.style.display = "inline-block"; confirmarBtn.onclick = () => confirmarVisita(); } /* ============================================================ 2) CONFIRMAR VISITA → WHATSAPP + EMAIL ============================================================ */ function confirmarVisita() { const nombre = document.getElementById("nombre").value; const telefono = document.getElementById("telefono").value; const referencia = document.getElementById("referencia").value; const cita = document.getElementById("citaElegida").value; const mensaje = `Ya has fijado la cita para la visita del inmueble ${referencia} (${cita}). El agente contactará con usted para confirmar o modificar la misma.`; // WhatsApp comprador if (telefono) { const waCliente = `https://wa.me/${telefono.replace(/[^0-9]/g, "")}?text=${encodeURIComponent(mensaje)}`; window.open(waCliente, "_blank"); } // WhatsApp agente const telefonoAgente = "34600000000"; // <-- CAMBIA ESTO const waAgente = `https://wa.me/${telefonoAgente}?text=${encodeURIComponent( `Cliente: ${nombre}\nTeléfono: ${telefono}\nInmueble: ${referencia}\nCita: ${cita}\n\n${mensaje}` )}`; window.open(waAgente, "_blank"); // Email agente const emailAgente = "tuemail@ejemplo.com"; // <-- CAMBIA ESTO const asunto = encodeURIComponent(`Cita fijada – Inmueble ${referencia}`); const cuerpo = encodeURIComponent( `Cliente: ${nombre}\nTeléfono: ${telefono}\nInmueble: ${referencia}\nCita: ${cita}\n\n${mensaje}` ); window.location.href = `mailto:${emailAgente}?subject=${asunto}&body=${cuerpo}`; alert("Cita registrada. El agente contactará para confirmar o modificar la misma."); document.getElementById("confirmarBtn").style.display = "none"; }
